```mermaid
// 用户实体类（存储用户核心数据）
class UserEntity {
  - userId: Integer       // 用户唯一ID（主键）
  - username: String      // 登录用户名（唯一）
  - password: String      // 加密后的密码
  - nickname: String      // 昵称
  - avatar: String        // 头像路径
  - status: Integer       // 账号状态（1-正常，0-封禁）
  - createTime: Date      // 注册时间
  - signature: String     // 个性签名
  - selfIntroduction: String // 自我介绍
  + getUserId(): Integer  // getter方法
  + setPassword(password: String): void // setter方法
}

// 登录请求数据传输类（接收前端参数）
class LoginDto {
  - username: String      // 登录用户名
  - password: String      // 登录密码
  + getUsername(): String
  + getPassword(): String
}

// 注册请求数据传输类（接收前端参数）
class RegisterDto {
  - username: String      // 注册用户名
  - password: String      // 密码
  - confirmPassword: String // 确认密码
  - nickname: String      // 昵称
  + getUsername(): String
  + getPassword(): String
}

// 控制器类（接收请求，返回响应）
class UserController {
  - userService: UserService
  + login(loginDto: LoginDto): Result // 登录接口
  + register(registerDto: RegisterDto): Result // 注册接口
  + getUserInfo(userId: Integer): UserVo // 获取用户信息
}

// 服务接口（定义业务逻辑规范）
interface UserService {
  + login(username: String, password: String): UserEntity // 登录校验
  + register(registerDto: RegisterDto): Boolean // 注册逻辑
  + getUserById(userId: Integer): UserEntity // 根据ID查询用户
}

// 服务实现类（实现业务逻辑）
class UserServiceImpl implements UserService {
  - userMapper: UserMapper
  + login(username: String, password: String): UserEntity // 比对密码，返回用户信息
  + register(registerDto: RegisterDto): Boolean // 密码加密，保存用户
  + getUserById(userId: Integer): UserEntity // 调用Mapper查询
}

// 数据访问接口（操作数据库）
interface UserMapper {
  + selectByUsername(username: String): UserEntity // 按用户名查询
  + insertUser(userEntity: UserEntity): Integer // 插入用户数据
  + selectById(userId: Integer): UserEntity // 按ID查询
}
```
