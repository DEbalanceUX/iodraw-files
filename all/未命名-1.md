```mermaid

    %% 用户实体类（存储用户核心数据）
    class UserEntity {
        - Integer userId       // 用户唯一ID（主键）
        - String username      // 登录用户名（唯一）
        - String password      // 加密后的密码
        - String nickname      // 昵称
        - String avatar        // 头像路径
        - Integer status       // 账号状态（1-正常，0-封禁）
        - Date createTime      // 注册时间
        - String signature     // 个性签名
        - String selfIntroduction // 自我介绍
        + Integer getUserId()  // getter方法
        + void setPassword(String password) // setter方法
    }

    %% 登录请求数据传输类（接收前端参数）
    class LoginDto {
        - String username      // 登录用户名
        - String password      // 登录密码
        + String getUsername()
        + String getPassword()
    }

    %% 注册请求数据传输类（接收前端参数）
    class RegisterDto {
        - String username      // 注册用户名
        - String password      // 密码
        - String confirmPassword // 确认密码
        - String nickname      // 昵称
        + String getUsername()
        + String getPassword()
    }

    %% 控制器类（接收请求，返回响应）
    class UserController {
        - UserService userService
        + Result login(LoginDto loginDto) // 登录接口
        + Result register(RegisterDto registerDto) // 注册接口
        + UserVo getUserInfo(Integer userId) // 获取用户信息
    }

    %% 服务接口（定义业务逻辑规范）
    interface UserService {
        + UserEntity login(String username, String password) // 登录校验
        + Boolean register(RegisterDto registerDto) // 注册逻辑
        + UserEntity getUserById(Integer userId) // 根据ID查询用户
    }

    %% 服务实现类（实现业务逻辑）
    class UserServiceImpl {
        - UserMapper userMapper
        + UserEntity login(String username, String password) // 比对密码，返回用户信息
        + Boolean register(RegisterDto registerDto) // 密码加密，保存用户
        + UserEntity getUserById(Integer userId) // 调用Mapper查询
    }

    %% 数据访问接口（操作数据库）
    interface UserMapper {
        + UserEntity selectByUsername(String username) // 按用户名查询
        + Integer insertUser(UserEntity userEntity) // 插入用户数据
        + UserEntity selectById(Integer userId) // 按ID查询
    }
```
