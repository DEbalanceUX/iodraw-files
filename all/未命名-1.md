    ArticleServiceImpl ..|> ArticleService
    ArticleEntity "n" -- "n" TagEntity : via ArticleTagEntity

    ArticleController ..> ArticlePublishDto
    ArticleServiceImpl ..> ArticleMapper
    ArticleServiceImpl ..> ArticleTagMapper
    ArticleServiceImpl ..> CategoryMapper
    ArticleService ..> ArticleEntity
    ArticleMapper ..> ArticleEntity
    ArticleTagMapper ..> ArticleTagEntity

    %% 核心类定义
    class ArticleEntity {
        - Integer articleId
        - String title
        - String content
        - Integer authorId
        - Integer categoryId
        - Integer status
        - Date publishedTime
        - Integer pageView
        - Integer commentCount
        + Integer getArticleId()
        + void setStatus(Integer status)
    }

    class ArticleTagEntity {
        - Integer middleId
        - Integer articleId
        - Integer tagId
        + Integer getArticleId()
        + Integer getTagId()
    }

    class ArticlePublishDto {
        - String title
        - String content
        - Integer categoryId
        - List<Integer> tagIds
        - Boolean isDraft
        + String getTitle()
        + String getContent()
    }

    class ArticleController {
        - ArticleService articleService
        + Result publishArticle(ArticlePublishDto dto, Integer userId)
        + Result saveDraft(ArticlePublishDto dto, Integer userId)
        + Result updateArticle(Integer articleId, ArticlePublishDto dto)
    }

    interface ArticleService {
        + Integer publish(ArticleEntity article, List<Integer> tagIds)
        + Integer saveDraft(ArticleEntity article)
        + Boolean updateArticle(ArticleEntity article, List<Integer> tagIds)
    }

    class ArticleServiceImpl {
        - ArticleMapper articleMapper
        - ArticleTagMapper articleTagMapper
        - CategoryMapper categoryMapper
        + Integer publish(ArticleEntity article, List<Integer> tagIds)
        + Integer saveDraft(ArticleEntity article)
    }

    interface ArticleMapper {
        + Integer insertArticle(ArticleEntity article)
        + Integer updateArticle(ArticleEntity article)
        + ArticleEntity selectById(Integer articleId)
    }

    interface ArticleTagMapper {
        + Integer batchInsert(List<ArticleTagEntity> tagEntities)
        + Integer deleteByArticleId(Integer articleId)
    }

    interface CategoryMapper {
        + Boolean checkCategoryValid(Integer categoryId)
    }

    class TagEntity {
        - Integer tagId
        - String tagName
        + Integer getTagId()
    }
